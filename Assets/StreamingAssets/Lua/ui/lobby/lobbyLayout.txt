local lobbyLayout = class("lobbyLayout", layout)

-- 一些定义
event_define.LobbyLayout_SHOW = "LobbyLayout_SHOW"
event_define.LobbyLayout_CLOSE = "LobbyLayout_CLOSE"

function lobbyLayout:ctor( id )
	lobbyLayout.super.ctor(self, id)	

    self:addEvent({ name = event_define.LobbyLayout_SHOW, eventHandler = lobbyLayout.super.Show})
    self:addEvent({ name = event_define.LobbyLayout_CLOSE, eventHandler = lobbyLayout.super.Close})
end	

function lobbyLayout:OnInit(event)
     -- 获得UI对象
    local btn_fight = self:ChildGameObject("Content/BtnFight") -- 对战
    local btn_rank = self:ChildGameObject("Content/BtnRank") -- 排位
    local btn_arena = self:ChildGameObject("Content/BtnArena") -- 竞技场
    local btn_risk = self:ChildGameObject("Content/BtnRisk") -- 冒险
    local btn_buy = self:ChildGameObject("Content/TopFrame/BtnBug") -- 购物
    local btn_pack = self:ChildGameObject("Content/ButtonFrame/BtnPack") -- 背包
    local btn_strategy = self:ChildGameObject("Content/ButtonFrame/BtnStrategy") -- 战略
    local btn_technology = self:ChildGameObject("Content/ButtonFrame/BtnTechnology") -- 科技
    local btn_arms = self:ChildGameObject("Content/ButtonFrame/BtnArms") -- 兵种
    local btn_general = self:ChildGameObject("Content/ButtonFrame/BtnGeneral") -- 将军
    local btn_join_team = self:ChildGameObject("Content/TopFrame/BtnJoinTeam") -- 加入军团
    -- 绑定事件
    UIEventListener.Get(btn_fight).onClick = function()
        findBattleDataManager:Instance():FindBattle(function () 
            print("加入对列")
            eventManager.postEvent({name = event_define.FindBattleLayout_SHOW})
        end)
        -- LuaUtils.SwitchToLevel(sceneManager.SceneId.scn_battle
    end
    UIEventListener.Get(btn_rank).onClick = self.FunctionLock
    UIEventListener.Get(btn_arena).onClick = self.FunctionLock
    UIEventListener.Get(btn_risk).onClick = self.FunctionLock
    UIEventListener.Get(btn_buy).onClick = self.FunctionLock
    UIEventListener.Get(btn_pack).onClick = self.FunctionLock
    UIEventListener.Get(btn_strategy).onClick = self.FunctionLock
    UIEventListener.Get(btn_technology).onClick = self.FunctionLock
    UIEventListener.Get(btn_arms).onClick = self.FunctionLock
    UIEventListener.Get(btn_general).onClick = self.FunctionLock
    UIEventListener.Get(btn_join_team).onClick = self.FunctionLock
end

function lobbyLayout:OnShow(event)
end

function lobbyLayout:OnHide(event)
end

function lobbyLayout:OnDestory( )
end

function lobbyLayout:FunctionLock()
    print("敬请期待")
end

return lobbyLayout
